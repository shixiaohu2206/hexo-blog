// 解法一
var map = {}
var head = 0
var i = 0
var max = 0
for (i; i < s.length; i++) {
  if (map[s[i]] >= head) {
    head = map[s[i]] + 1
  }
  map[s[i]] = i
  max = max > i - head + 1 ? max : i - head + 1
}
return max

const map = {}
let left = 0
let max = s.split('').reduce((max, v, i) => {
  /*
   * map[v] 初始为underfined，表明s[i]还没出现过，
   * 一旦map[v]有值了，且大于left，就将left的值设为map[v]后一位
   */
  left = map[v] >= left ? map[v] + 1 : left
  map[v] = i
